C Distance Indicators

本戦のB問題もこういう感じの問題だったので，反省を生かしてループを減らしていきたい．絶対にmapと二分探索を使うという固い意志．

とりあえず式は変形してAi+i=j-Ajになる．右辺をvectorで，左辺をmap<val,it>で管理する．

右辺を順番に見て，値がmapに保管されていたら，そのjからnまでの個数が答えになる．

 1int main() {
 2	int n; cin >> n;
 3	vector<lint> a(n);
 4	rep(i, 0, n) cin >> a[i];
 5	//Ai+i
 6	vector<lint> Ai(n);
 7	rep(i, 0, n) Ai[i] = a[i] + (i+1);
 8	//j-Aj
 9	map<lint, set<int>> mp;
10	rep(i, 0, n) mp[(i + 1) - a[i]].insert(i);
11
12	lint ans = 0;
13	for(int i = 0; i < n; i++) {
14		if(mp.find(Ai[i]) != mp.end()) {
15			auto it = mp[Ai[i]].lower_bound(i);
16			if (it != mp[Ai[i]].end()) {
17				ans += mp[Ai[i]].size() - distance(mp[Ai[i]].begin(), it);
18			}
19		}
20	}
21	cout << ans << endl;
22	return 0;
23}

E A Path in A Dictionary

辞書順最小はGreedyだってけんちょんさんが言ってたので，そうなんだと思う．普段の順路検索は配列の追加順にやっているので，これを文字でソートすればAからZの順に探索をしてくれる．そうすれば外側から自動的に小さい単語から埋まっていく．やるのはソートと文字列（経路）保存だけ！はじめvectorに経路をいれていったが，反転のところがよくなさそうだった．こういう時はstackを使えばいい．

13bool dfs(const Graph& G, int x, int y) {
14	//経路追加
15	ans.emplace_back(x);
16	//x==yならおわり
17	if (x == y) return true;
18
19	//いつものdfs
20	//訪問済み
21	seen[x] = true;
22	for (auto& v : G[x]) {
23		if (!seen[v]) {
24			if (dfs(G, v, y)) return true;
25		}
26	}
27	//次に行けなかったので経路から取り出す
28	ans.pop_back();
29	return false;
30}
